    function move() {
      x += dx;
      y += dy;

      let bounced = false;

      if (x + imgWidth >= window.innerWidth || x <= 0) {
        dx = (dx > 0 ? -1 : 1) * (Math.random() * 2 + 1);
        bounced = true;
      }

      if (y + imgHeight >= window.innerHeight || y <= 0) {
        dy = (dy > 0 ? -1 : 1) * (Math.random() * 2 + 1);

      }

      }

      img.style.left = x + "px";
      img.style.top = y + "px";
